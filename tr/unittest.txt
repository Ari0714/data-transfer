import unittest
from unittest.mock import MagicMock, patch
import json
import sys

# Assuming your original source code file is named adls_connector.py
from adls_connector import ADLSConnector


class TestADLSConnector(unittest.TestCase):

    def setUp(self):
        # Reset the singleton instance before each test to ensure test isolation
        ADLSConnector._instance = None

        # Prepare mock objects for SparkSession and file path
        self.mock_spark = MagicMock()
        self.file_path = "abfss://mycontainer@myaccount.dfs.core.windows.net/mydir"

        # Mock environment variables required by the connector
        self.env_patcher = patch.dict('os.environ', {
            'CATALOG_URL': 'http://mock-catalog',
            'ACCOUNT': 'myaccount',
            'CONTAINER': 'mycontainer',
            'DIR': 'mydir',
            'REST_TOKEN': 'mock_token'
        })
        self.env_patcher.start()

    def tearDown(self):
        # Clean up the environment patcher and reset the singleton status
        self.env_patcher.stop()
        ADLSConnector._instance = None

    @patch('adls_connector.ADLSConnector._authenticate')
    def test_singleton_pattern(self, mock_auth):
        """Verify that the singleton pattern returns the identical instance."""
        mock_auth.return_value = MagicMock()

        instance1 = ADLSConnector(self.mock_spark, self.file_path)
        instance2 = ADLSConnector(self.mock_spark, self.file_path)

        # Both instances must point to the exact same memory address
        self.assertIs(instance1, instance2)

    @patch('adls_connector.urllib.request.urlopen')
    @patch('adls_connector.urllib.request.Request')
    @patch('adls_connector.ADLSConnector._authenticate')
    def test_post_json(self, mock_auth, mock_request, mock_urlopen):
        """Verify that post_json accurately sends payloads and parses JSON responses."""
        mock_auth.return_value = MagicMock()

        # Mock the network stream returning a JSON byte string
        mock_response = MagicMock()
        mock_response.read.return_value = b'{"status": "success"}'
        mock_urlopen.return_value = mock_response

        connector = ADLSConnector(self.mock_spark, self.file_path)

        url = "http://test-url"
        payload = {"key": "value"}
        token = "test_token"

        result = connector.post_json(url, payload, token)

        # Assert that the parsed dictionary matches the expected value
        self.assertEqual(result, {"status": "success"})
        # Validate that the urllib Request object was correctly initialized
        mock_request.assert_called_once()

    @patch('adls_connector.ADLSConnector._authenticate')
    def test_find_sas_recursive(self, mock_auth):
        """Test the recursive behavior of _find_sas when traversing complex structures."""
        mock_auth.return_value = MagicMock()

        connector = ADLSConnector(self.mock_spark, self.file_path)

        # Deeply nested data structure containing the target SAS Token at the leaf node
        nested_obj = {
            "level1": [
                {"not_me": "random"},
                {"target": "something?sv=2021&sig=abcdefg"}
            ]
        }

        sas = connector._find_sas(nested_obj)
        self.assertEqual(sas, "sv=2021&sig=abcdefg")

    @patch('adls_connector.DataLakeServiceClient')
    @patch('adls_connector.ADLSConnector.post_json')
    def test_authenticate_success(self, mock_post_json, mock_service_client):
        """Verify successful token acquisition and correct FileSystemClient routing."""
        mock_fs_client = MagicMock()
        mock_svc_instance = MagicMock()
        mock_svc_instance.get_file_system_client.return_value = mock_fs_client
        mock_service_client.return_value = mock_svc_instance

        # Mock post_json returning valid credential payloads prior to object creation
        mock_post_json.return_value = {
            'credentials': {
                'adls.sas-token.myaccount': 'mock_sas_token_value',
                'adls.sas-token.expires-at-ms.myaccount': '1700000000000'
            }
        }

        # Instantiation will trigger _authenticate, which internally runs our mocked post_json safely
        connector = ADLSConnector(self.mock_spark, self.file_path)

        # Ensure the assigned file system client matches our mocked instance
        self.assertEqual(connector.fs, mock_fs_client)
        mock_service_client.assert_called_once_with(
            'https://myaccount.dfs.core.windows.net',
            credential='mock_sas_token_value',
            connection_verify=False
        )

    @patch('adls_connector.ADLSConnector._authenticate')
    def test_parse_adls_path_valid(self, mock_auth):
        """Verify that _parse_adls_path correctly splits structured abfss:// paths."""
        mock_auth.return_value = MagicMock()
        connector = ADLSConnector(self.mock_spark, self.file_path)

        test_path = "abfss://container1@account1.dfs.core.windows.net/folder/file.txt"
        container, account, key = connector._parse_adls_path(test_path)

        self.assertEqual(container, "container1")
        self.assertEqual(account, "account1")
        self.assertEqual(key, "folder/file.txt")

    @patch('adls_connector.ADLSConnector._authenticate')
    def test_parse_adls_path_invalid(self, mock_auth):
        """Verify that _parse_adls_path raises a ValueError when dealing with malformed paths."""
        mock_auth.return_value = MagicMock()
        connector = ADLSConnector(self.mock_spark, self.file_path)

        invalid_path = "https://invalid-scheme.com/file"
        with self.assertRaises(ValueError):
            connector._parse_adls_path(invalid_path)

    def test_load_script(self):
        """Verify that load_script downloads and successfully decodes remote script content."""
        mock_fs = MagicMock()
        mock_file_client = MagicMock()
        mock_download = MagicMock()

        # Mock the stream response representing remote data payload
        mock_download.readall.return_value = b"print('hello')"
        mock_file_client.download_file.return_value = mock_download
        mock_fs.get_file_client.return_value = mock_file_client

        with patch.object(ADLSConnector, '_authenticate', return_value=mock_fs):
            connector = ADLSConnector(self.mock_spark, self.file_path)

            content = connector.load_script("abfss://c@a.dfs.core.windows.net/script.py")
            self.assertEqual(content, "print('hello')")
            mock_fs.get_file_client.assert_called_with("script.py")

    @patch('builtins.open', new_callable=unittest.mock.mock_open)
    def test_download_file(self, mock_file_open):
        """Verify that download_file writes downloaded bytes to the local file system."""
        mock_fs = MagicMock()
        mock_file_client = MagicMock()
        mock_download = MagicMock()

        mock_download.readall.return_value = b"binary_data"
        mock_file_client.download_file.return_value = mock_download
        mock_fs.get_file_client.return_value = mock_file_client

        with patch.object(ADLSConnector, '_authenticate', return_value=mock_fs):
            connector = ADLSConnector(self.mock_spark, self.file_path)

            local_path = "/local/path/file.txt"
            result = connector.download_file("abfss://c@a.dfs.core.windows.net/file.txt", local_path)

            self.assertEqual(result, local_path)
            # Ensure local safe I/O context opens the correct path with binary write permissions
            mock_file_open.assert_called_once_with(local_path, "wb")
            mock_file_open().write.assert_called_once_with(b"binary_data")


if __name__ == '__main__':
    unittest.main()
