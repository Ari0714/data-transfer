class TestADLSConnector(unittest.TestCase):
    """Unit tests for ADLSConnector class"""

    def setUp(self):
        """Set up test fixtures before each test method"""
        # Mock SparkSession
        self.mock_spark = Mock(spec=SparkSession)

        # Set environment variables for testing
        self.env_patcher = patch.dict('os.environ', {
            'CATALOG_URL': 'http://test-catalog:8181',
            'ACCOUNT': 'testaccount',
            'CONTAINER': 'testcontainer',
            'REST_TOKEN': 'test-token-123',
            'DIR': 'test_base_path'
        })
        self.env_patcher.start()

        # Test file path
        self.test_file_path = "abfas://testcontainer@testaccount.dfs.core.windows.net/test/path/file.parquet"

        # Create connector instance
        self.connector = ADLSConnector(self.mock_spark, self.test_file_path)

    def tearDown(self):
        """Clean up after each test method"""
        self.env_patcher.stop()
        ADLSConnector.instance = None

    # ========== Tests for __new__ and __init__ ==========

    def test_singleton_pattern(self):
        """Test that ADLSConnector follows singleton pattern"""
        connector2 = ADLSConnector(self.mock_spark, self.test_file_path)
        self.assertIs(self.connector, connector2)

    def test_init_skips_if_already_initialized(self):
        """Test that __init__ skips initialization if already done"""
        # Set adls_client to a mock value
        self.connector.adls_client = Mock()
        mock_fs = Mock()
        self.connector.fs = mock_fs

        # Re-initialize
        self.connector.__init__(self.mock_spark, self.test_file_path)

        # fs should remain unchanged
        self.assertEqual(self.connector.fs, mock_fs)

    # ========== Tests for post_json ==========

    @patch('urllib.request.urlopen')
    def test_post_json_success_without_token(self, mock_urlopen):
        """Test post_json successfully posts JSON without token"""
        # Mock response
        mock_response = Mock()
        mock_response.read.return_value = json.dumps({"key": "value"}).encode()
        mock_urlopen.return_value = mock_response

        url = "http://test.com/api"
        payload = {"test": "data"}
        result = self.connector.post_json(url, payload)

        self.assertEqual(result, {"key": "value"})
        mock_urlopen.assert_called_once()

    @patch('urllib.request.urlopen')
    def test_post_json_success_with_token(self, mock_urlopen):
        """Test post_json successfully posts JSON with bearer token"""
        mock_response = Mock()
        mock_response.read.return_value = json.dumps({"key": "value"}).encode()
        mock_urlopen.return_value = mock_response

        url = "http://test.com/api"
        payload = {"test": "data"}
        token = "my-bearer-token"
        result = self.connector.post_json(url, payload, token)

        self.assertEqual(result, {"key": "value"})

        # Verify authorization header was set
        call_args = mock_urlopen.call_args[0][0]
        headers = call_args.headers
        self.assertEqual(headers['Authorization'], f'Bearer {token}')

    # ========== Tests for _find_sas ==========

    def test_find_sas_in_string_with_sas_params(self):
        """Test find_sas extracts SAS token from string containing sv= and sig="""
        sas_string = "?sv=2022-11-02&sig=abc123&se=2024-01-01"
        result = self.connector._find_sas(sas_string)
        self.assertEqual(result, "sv=2022-11-02&sig=abc123&se=2024-01-01")

    def test_find_sas_in_string_without_question_mark(self):
        """Test find_sas handles SAS string without leading question mark"""
        sas_string = "sv=2022-11-02&sig=abc123&se=2024-01-01"
        result = self.connector._find_sas(sas_string)
        self.assertEqual(result, "sv=2022-11-02&sig=abc123&se=2024-01-01")

    def test_find_sas_in_string_no_sas_params(self):
        """Test find_sas returns None when no SAS params found"""
        sas_string = "regular string without sas"
        result = self.connector._find_sas(sas_string)
        self.assertIsNone(result)

    def test_find_sas_in_dict(self):
        """Test find_sas recursively searches dictionary for SAS token"""
        test_dict = {
            "key1": "value1",
            "key2": {
                "nested": "?sv=2022&sig=xyz789"
            }
        }
        result = self.connector._find_sas(test_dict)
        self.assertEqual(result, "sv=2022&sig=xyz789")

    def test_find_sas_in_list(self):
        """Test find_sas recursively searches list for SAS token"""
        test_list = [
            "normal string",
            {"nested": "?sv=2023&sig=abc456"}
        ]
        result = self.connector._find_sas(test_list)
        self.assertEqual(result, "sv=2023&sig=abc456")

    def test_find_sas_no_match(self):
        """Test find_sas returns None when no SAS token found in complex structure"""
        test_data = {
            "key1": "value1",
            "key2": [1, 2, 3],
            "key3": {"nested": "no sas here"}
        }
        result = self.connector._find_sas(test_data)
        self.assertIsNone(result)

    # ========== Tests for _parse_adls_path ==========

    def test_parse_valid_abfs_path(self):
        """Test _parse_adls_path correctly parses valid abfs:// path"""
        path = "abfs://mycontainer@myaccount.dfs.core.windows.net/folder/subfolder/file.txt"
        container, account, key = self.connector._parse_adls_path(path)

        self.assertEqual(container, "mycontainer")
        self.assertEqual(account, "myaccount")
        self.assertEqual(key, ["folder", "subfolder", "file.txt"])

    def test_parse_valid_abfas_path(self):
        """Test _parse_adls_path correctly parses valid abfas:// path"""
        path = "abfas://mycontainer@myaccount.dfs.core.windows.net/folder/file.csv"
        container, account, key = self.connector._parse_adls_path(path)

        self.assertEqual(container, "mycontainer")
        self.assertEqual(account, "myaccount")
        self.assertEqual(key, ["folder", "file.csv"])

    def test_parse_path_without_scheme(self):
        """Test _parse_adls_path raises error for invalid scheme"""
        path = "http://mycontainer@myaccount.dfs.core.windows.net/folder/file.txt"
        with self.assertRaises(ValueError) as context:
            self.connector._parse_adls_path(path)
        self.assertIn("Invalid ADLS path scheme", str(context.exception))

    def test_parse_path_missing_netloc(self):
        """Test _parse_adls_path raises error when netloc is missing"""
        path = "abfs:///folder/file.txt"
        with self.assertRaises(ValueError) as context:
            self.connector._parse_adls_path(path)
        self.assertIn("Expected format", str(context.exception))

    def test_parse_path_invalid_split(self):
        """Test _parse_adls_path raises error when container/account split fails"""
        path = "abfs://invalidformat/path/file.txt"
        with self.assertRaises(ValueError) as context:
            self.connector._parse_adls_path(path)
        self.assertIn("Path split failed", str(context.exception))

    def test_parse_path_allows_empty_path(self):
        """Test _parse_adls_path allows empty path when allow_empty_path=True"""
        path = "abfs://mycontainer@myaccount.dfs.core.windows.net/"
        container, account, key = self.connector._parse_adls_path(path, allow_empty_path=True)
        self.assertEqual(container, "mycontainer")
        self.assertEqual(account, "myaccount")
        self.assertEqual(key, [])
